42  struct NameTopoKeyCompare
43  {
44  bool operator()(const NameTopoKey &lhs, const NameTopoKey &rhs) const
45  {
46  assert(lhs.second != nullptr);
47  assert(rhs.second != nullptr);
48  return lhs.first < rhs.first ||
49  (!(rhs.first < lhs.first) && lhs.second->name() < rhs.second->name());
50  }
51  };

56  {
