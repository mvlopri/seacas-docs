32  /** \brief Base class for all 'grouping' entities.
33  * The following derived classes are typical:
34  *
35  * -- NodeSet -- grouping of nodes (0d topology)
36  *
37  * -- EdgeSet -- grouping of edges (1d topology)
38  *
39  * -- FaceSet -- grouping of faces (2d topology) [Surface]
40  *
41  * Similarly, there is:
42  *
43  * -- NodeBlock -- grouping of 'similar' nodes (same degree of freedom, ...)
44  *
45  * -- ElementBlock -- grouping of 'similar' elements (same element topology,
46  * attributes, ...)
47  * 0d, 1d, 2d, 3d topology possible -- e.g., sphere, bar, quad, hex
48  *
49  * A Region is also a grouping entity, except that its list of subentites
50  * are other GroupingEntities. That is, it maintains a list of NodeBlocks,
51  * ElementBlocks, NodeLists, CommLists and Surfaces. [Similar to the
52  * "Composite Patter" in Design Patterns] All interface to GroupingEntities
53  * is through the Region class; clients of the IO subsystem have no direct
54  * access to the underlying GroupingEntities (other than the Region).
55  *
56  * Each GroupingEntity contains:
57  *
58  * -- name
59  *
60  * -- MeshEntities of the specified topological dimension
61  *
62  * -- Optional attributes, either global (applied to the groupingentity), or
63  * unique value(s) to be applied to each subentity.
64  *
65  * -- Data items
66  */
67  class GroupingEntity

457 /** \brief Read type 'T' field data from the database file into memory using a std::vector.
458  *
459  * \param[in] field_name The name of the field to read.
460  * \param[out] data The data.
461  * \returns The number of values read.
462  *
463  */
464 template <typename T>
465 int64_t Ioss::GroupingEntity::get_field_data(const std::string &field_name,
466  std::vector<T> & data) const
467 {
468  verify_field_exists(field_name, "input");
469 
470  Ioss::Field field = get_field(field_name);
471  field.check_type(Ioss::Field::get_field_type(T(0)));
472 
473  data.resize(field.raw_count() * field.raw_storage()->component_count());
474  size_t data_size = data.size() * sizeof(T);
475  auto retval = internal_get_field_data(field, data.data(), data_size);
476 
477  // At this point, transform the field if specified...
478  if (retval >= 0) {
479  field.transform(data.data());
480  }
481 
482  return retval;
483 }

485 /** \brief Write type 'T' field data from memory into the database file using a std::vector.
486  *
487  * \param[in] field_name The name of the field to write.
488  * \param[in] data The data.
489  * \returns The number of values written.
490  *
491  */
492 template <typename T>
493 int64_t Ioss::GroupingEntity::put_field_data(const std::string & field_name,
494  const std::vector<T> &data) const
495 {
496  verify_field_exists(field_name, "output");
497 
498  Ioss::Field field = get_field(field_name);
499  field.check_type(Ioss::Field::get_field_type(T(0)));
500  size_t data_size = data.size() * sizeof(T);
501  if (field.has_transform()) {
502  // Need non-const data since the transform will change the users data.
503  std::vector<T> nc_data(data);
504  field.transform(nc_data.data());
505  return internal_put_field_data(field, nc_data.data(), data_size);
506  }
507 
508  T *my_data = const_cast<T *>(data.data());
509  return internal_put_field_data(field, my_data, data_size);
510 }

528 /** \brief Read field data from the database file into memory using a 1-D Kokkos:::View.
529  *
530  * \tparam T The data type.
531  * \tparam Args The other template arguments for data.
532  * \param[in] field_name The name of the field to read.
533  * \param[out] data The data.
534  * \returns The number of values read.
535  *
536  */
537 template <typename T, typename... Args>
538 int64_t Ioss::GroupingEntity::get_field_data(const std::string & field_name,
539  Kokkos::View<T *, Args...> &data) const
540 {
541  typedef Kokkos::View<T *, Args...> ViewType;
542 
543  verify_field_exists(field_name, "input");
544 
545  Ioss::Field field = get_field(field_name);
546 
547  // Resize the view
548  auto new_view_size = field.raw_count() * field.raw_storage()->component_count();
549  Kokkos::resize(data, new_view_size);
550  size_t data_size = new_view_size * sizeof(T);
551 
552  // Create a host mirror view. (No memory allocation if data is in HostSpace.)
553  typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
554 
555  // Extract a pointer to the underlying allocated memory of the host view.
556  T *host_data_ptr = host_data.data();
557 
558  // Extract the data from disk to the underlying memory pointed to by host_data_ptr.
559  auto retval = internal_get_field_data(field, host_data_ptr, data_size);
560 
561  // At this point, transform the field if specified...
562  if (retval >= 0)
563  field.transform(host_data_ptr);
564 
565  // Copy the data to the device. (No op if data is in HostSpace.)
566  Kokkos::deep_copy(data, host_data);
567 
568  return retval;
569 }

571 /** \brief Read field data from the database file into memory using a 2-D Kokkos:::View.
572  *
573  * \tparam T The data type
574  * \tparam Args The other template arguments for data.
575  * \param[in] field_name The name of the field to read.
576  * \param[out] data The data.
577  * \returns The number of values read.
578  *
579  */
580 template <typename T, typename... Args>
581 int64_t Ioss::GroupingEntity::get_field_data(const std::string & field_name,
582  Kokkos::View<T **, Args...> &data) const
583 {
584  typedef Kokkos::View<T **, Args...> ViewType;
585 
586  verify_field_exists(field_name, "input");
587 
588  Ioss::Field field = get_field(field_name);
589 
590  // Resize the view
591  int new_view_size_left = field.raw_count();
592  int new_view_size_right = field.raw_storage()->component_count();
593  Kokkos::resize(data, new_view_size_left, new_view_size_right);
594  size_t data_size = new_view_size_left * new_view_size_right * sizeof(T);
595 
596  // Create and allocate an array to hold the data temporarily.
597  // This is necessary to ensure the data is placed in the correct
598  // location in the 2-D array, avoiding incorrect placement due
599  // to Views with padded dimensions.
600  T *data_array = new T[data_size];
601 
602  // Create a host mirror view. (No memory allocation if data is in HostSpace.)
603  typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
604 
605  // Extract the data from disk to the underlying memory pointed to by host_data_ptr.
606  auto retval = internal_get_field_data(field, data_array, data_size);
607 
608  // At this point, transform the field if specified...
609  if (retval >= 0)
610  field.transform(data_array);
611 
612  // Copy the data to the host Mirror view.
613  // The host mirror view has the same layout as the device view.
614  // For CUDA, this will be LayoutLeft. In this case, the loop order
615  // chosen here will be slower than the reversed loop order.
616  // However, The time for this extra in-memory copy is small
617  // compared with the time to copy from disk into memory.
618  for (int i = 0; i < new_view_size_left; ++i) {
619  for (int j = 0; j < new_view_size_right; ++j) {
620  host_data(i, j) = data_array[new_view_size_right * i + j];
621  }
622  }
623 
624  // Delete the temporary array
625  delete[] data_array;
626 
627  // Copy the data to the device. (No op if data is in HostSpace.)
628  Kokkos::deep_copy(data, host_data);
629 
630  return retval;
631 }

633 /** \brief Write field data from memory into the database file using a 1-D Kokkos::View.
634  *
635  * \tparam T The data type
636  * \tparam Args The other template arguments for data.
637  * \param[in] field_name The name of the field to write.
638  * \param[in] data The data.
639  * \returns The number of values written.
640  *
641  */
642 template <typename T, typename... Args>
643 int64_t Ioss::GroupingEntity::put_field_data(const std::string & field_name,
644  Kokkos::View<T *, Args...> &data) const
645 {
646  typedef Kokkos::View<T *, Args...> ViewType;
647 
648  verify_field_exists(field_name, "output");
649 
650  Ioss::Field field = get_field(field_name);
651  size_t data_size = field.raw_count() * field.raw_storage()->component_count() * sizeof(T);
652 
653  // Create a host mirror view. (No memory allocation if data is in HostSpace.)
654  typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
655 
656  // Copy the data to the host. (No op if data is in HostSpace.)
657  Kokkos::deep_copy(host_data, data);
658 
659  // Extract a pointer to the underlying allocated memory of the host view.
660  T *host_data_ptr = host_data.data();
661 
662  // Transform the field
663  field.transform(host_data_ptr);
664 
665  // Copy the data to disk from the underlying memory pointed to by host_data_ptr.
666  return internal_put_field_data(field, host_data_ptr, data_size);
667 }

669 /** \brief Write field data from memory into the database file using a 2-D Kokkos::View.
670  *
671  * \tparam T The data type
672  * \tparam Args The other template arguments for data.
673  * \param[in] field_name The name of the field to write.
674  * \param[in] data The data.
675  * \returns The number of values written.
676  *
677  */
678 template <typename T, typename... Args>
679 int64_t Ioss::GroupingEntity::put_field_data(const std::string & field_name,
680  Kokkos::View<T **, Args...> &data) const
681 {
682  typedef Kokkos::View<T **, Args...> ViewType;
683 
684  verify_field_exists(field_name, "output");
685 
686  Ioss::Field field = get_field(field_name);
687 
688  int view_size_left = data.extent(0);
689  int view_size_right = data.extent(1);
690  size_t data_size = field.raw_count() * field.raw_storage()->component_count() * sizeof(T);
691 
692  if (view_size_left * view_size_right * sizeof(T) != data_size) {
693  std::ostringstream errmsg;
694  errmsg << "\nERROR: View dimensions are inconsistent with field raw count or raw storage "
695  "component count"
696  << "for field" << field_name << "\n\n";
697  IOSS_ERROR(errmsg);
698  }
699 
700  // Create a host mirror view. (No memory allocation if data is in HostSpace.)
701  typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
702 
703  // Copy the data to the host. (No op if data is in HostSpace.)
704  Kokkos::deep_copy(host_data, data);
705 
706  // Create and allocate an array to hold the data temporarily.
707  // This is necessary to ensure the data is taken from the correct
708  // location in the 2-D array, avoiding incorrect location due
709  // to Views with padded dimensions.
710  T *data_array = new T[data_size];
711 
712  // Copy the data from the host Mirror view.
713  // The host mirror view has the same layout as the device view.
714  // For CUDA, this will be LayoutLeft. In this case, the loop order
715  // chosen here will be slower than the reversed loop order.
716  // However, The time for this extra in-memory copy is small
717  // compared with the time to copy to disk from memory.
718  for (int i = 0; i < view_size_left; ++i) {
719  for (int j = 0; j < view_size_right; ++j) {
720  data_array[view_size_right * i + j] = host_data(i, j);
721  }
722  }
723 
724  // Transform the field
725  field.transform(data_array);
726 
727  // Copy the data to disk from the underlying memory pointed to by data_array.
728  auto retval = internal_put_field_data(field, data_array, data_size);
729 
730  // Delete the temporary array
731  delete[] data_array;
732 
733  return retval;
734 }
