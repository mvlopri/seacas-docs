74  struct FaceEqual
75  {
76  bool operator()(const Face &left, const Face &right) const
77  {
78  if (left.hashId_ != right.hashId_) {
79  return false;
80  }
81  // Hash (hashId_) is equal
82  // Check whether same vertices (can be in different order)
83  // Most (All?) of the time, there are no hashId_ collisions, so this test will not
84  // find a difference and the function will return 'true'
85  // However, for some reason, removing this check does not change the execution time
86  // appreiciably...
87  for (auto lvert : left.connectivity_) {
88  if (std::find(right.connectivity_.cbegin(), right.connectivity_.cend(), lvert) ==
89  right.connectivity_.cend()) {
90  // Not found, therefore not the same.
91  return false;
92  }
93  }
94  return true;
95  }
96  };
