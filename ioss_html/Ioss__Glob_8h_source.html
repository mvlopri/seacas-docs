185 std::tuple<bool, size_t> ExecAux(const String<charT> &str, bool comp_end = true) const
186 {
187 size_t state_pos = 0;
188 size_t str_pos = 0;
189
190 // run the state vector until state reaches fail or match state, or
191 // until the string is all consumed
192 while (state_pos != fail_state_ && state_pos != match_state_ && str_pos < str.length()) {
193 std::tie(state_pos, str_pos) = states_[state_pos]->Next(str, str_pos);
194 }
195
196 // if comp_end is true it matches only if the automata reached the end of
197 // the string
198 if (comp_end) {
199 if ((state_pos == match_state_) && (str_pos == str.length())) {
200 return std::tuple<bool, size_t>(state_pos == match_state_, str_pos);
201 }
202
203 return std::tuple<bool, size_t>(false, str_pos);
204 }
205 else {
206 // if comp_end is false, compare only if the states reached the
207 // match state
208 return std::tuple<bool, size_t>(state_pos == match_state_, str_pos);
209 }
210 }

637 std::vector<Token<charT>> Scanner()
638 {
639 std::vector<Token<charT>> tokens;
640 while (true) {
641 switch (c_) {
642 case '?': {
643 Advance();
644 if (c_ == '(') {
645 tokens.push_back(Select(TokenKind::QUESTLPAREN));
646 Advance();
647 }
648 else {
649 tokens.push_back(Select(TokenKind::QUESTION));
650 }
651 break;
652 }
653
654 case '*': {
655 Advance();
656 if (c_ == '(') {
657 tokens.push_back(Select(TokenKind::STARLPAREN));
658 Advance();
659 }
660 else {
661 tokens.push_back(Select(TokenKind::STAR));
662 }
663 break;
664 }
665
666 case '+': {
667 Advance();
668 if (c_ == '(') {
669 tokens.push_back(Select(TokenKind::PLUSLPAREN));
670 Advance();
671 }
672 else {
673 tokens.push_back(Select(TokenKind::CHAR, '+'));
674 }
675 break;
676 }
677
678 case '-': {
679 tokens.push_back(Select(TokenKind::SUB));
680 Advance();
681 break;
682 }
683
684 case '|': {
685 tokens.push_back(Select(TokenKind::UNION));
686 Advance();
687 break;
688 }
689
690 case '@': {
691 Advance();
692 if (c_ == '(') {
693 tokens.push_back(Select(TokenKind::ATLPAREN));
694 Advance();
695 }
696 else {
697 tokens.push_back(Select(TokenKind::CHAR, '@'));
698 }
699 break;
700 }
701
702 case '!': {
703 Advance();
704 if (c_ == '(') {
705 tokens.push_back(Select(TokenKind::NEGLPAREN));
706 Advance();
707 }
708 else {
709 tokens.push_back(Select(TokenKind::CHAR, '!'));
710 }
711 break;
712 }
713
714 case '(': {
715 tokens.push_back(Select(TokenKind::LPAREN));
716 Advance();
717 break;
718 }
719
720 case ')': {
721 tokens.push_back(Select(TokenKind::RPAREN));
722 Advance();
723 break;
724 }
725
726 case '[': {
727 Advance();
728 if (c_ == '!') {
729 tokens.push_back(Select(TokenKind::NEGLBRACKET));
730 Advance();
731 }
732 else {
733 tokens.push_back(Select(TokenKind::LBRACKET));
734 }
735 break;
736 }
737
738 case ']': {
739 tokens.push_back(Select(TokenKind::RBRACKET));
740 Advance();
741 break;
742 }
743
744 case '\\': {
745 Advance();
746 if (c_ == kEndOfInput) {
747 throw Error("No valid char after '\\'");
748 }
749 else if (IsSpecialChar(c_)) {
750 tokens.push_back(Select(TokenKind::CHAR, c_));
751 Advance();
752 }
753 break;
754 }
755
756 default: {
757 if (c_ == kEndOfInput) {
758 tokens.push_back(Select(TokenKind::EOS));
759 return tokens;
760 }
761 else {
762 tokens.push_back(Select(TokenKind::CHAR, c_));
763 Advance();
764 }
765 }
766 }
767 }

1029 {
1030 public:
1031 Parser() = delete;

1035 AstNodePtr<charT> GenAst() { return ParserGlob(); }

1038 AstNodePtr<charT> ParserChar()
1039 {
1040 Token<charT> &tk = NextToken();
1041 if (tk != TokenKind::CHAR) {
1042 throw Error("char expected");
1043 }
1044
1045 charT c = tk.Value();
1046 return AstNodePtr<charT>(new CharNode<charT>(c));
1047 }
1048
1049 AstNodePtr<charT> ParserRange()
1050 {
1051 AstNodePtr<charT> char_start = ParserChar();
1052
1053 Token<charT> &tk = NextToken();
1054 if (tk != TokenKind::SUB) {
1055 throw Error("range expected");
1056 }
1057
1058 AstNodePtr<charT> char_end = ParserChar();

1060 }
1061
1062 AstNodePtr<charT> ParserSetItem()
1063 {
1064 if (PeekAhead() == TokenKind::SUB) {
1065 return ParserRange();
1066 }
1067
1068 return ParserChar();

1071 AstNodePtr<charT> ParserSetItems()
1072 {
1073 std::vector<AstNodePtr<charT>> items;
1074
1075 do {
1076 items.push_back(ParserSetItem());
1077 } while (GetToken() != TokenKind::RBRACKET);
1078
1079 Advance();
1080
1081 return AstNodePtr<charT>(new SetItemsNode<charT>(std::move(items)));
1082 }
1083
1084 AstNodePtr<charT> ParserSet()
1085 {
1086 Token<charT> &tk = NextToken();
1087
1088 if (tk == TokenKind::LBRACKET) {
1089 return AstNodePtr<charT>(new PositiveSetNode<charT>(ParserSetItems()));
1090 }
1091 else if (tk == TokenKind::NEGLBRACKET) {
1092 return AstNodePtr<charT>(new NegativeSetNode<charT>(ParserSetItems()));
1093 }
1094 else {
1095 throw Error("set expected");
1096 }
1097 }
1098
1099 AstNodePtr<charT> ParserBasicGlob()
1100 {
1101 Token<charT> &tk = GetToken();
1102
1103 switch (tk.Kind()) {
1104 case TokenKind::QUESTION: Advance(); return AstNodePtr<charT>(new AnyNode<charT>());
1105
1106 case TokenKind::STAR: Advance(); return AstNodePtr<charT>(new StarNode<charT>());
1107
1108 case TokenKind::SUB: Advance(); return AstNodePtr<charT>(new CharNode<charT>('-'));
1109
1110 case TokenKind::CHAR: return ParserChar();
1111
1112 case TokenKind::LBRACKET:
1113 case TokenKind::NEGLBRACKET: return ParserSet();
1114
1115 case TokenKind::LPAREN:
1116 case TokenKind::QUESTLPAREN:
1117 case TokenKind::STARLPAREN:
1118 case TokenKind::PLUSLPAREN:
1119 case TokenKind::NEGLPAREN:
1120 case TokenKind::ATLPAREN: return ParserGroup();
1121
1122 default: throw Error("basic glob expected");
1123 }
1124 }
1125
1126 AstNodePtr<charT> ParserGroup()
1127 {
1128 typename GroupNode<charT>::GroupType type;
1129 Token<charT> &tk = NextToken();
1130
1131 switch (tk.Kind()) {
1132 case TokenKind::LPAREN: type = GroupNode<charT>::GroupType::BASIC; break;
1133
1134 case TokenKind::QUESTLPAREN: type = GroupNode<charT>::GroupType::ANY; break;
1135
1136 case TokenKind::STARLPAREN: type = GroupNode<charT>::GroupType::STAR; break;
1137
1138 case TokenKind::PLUSLPAREN: type = GroupNode<charT>::GroupType::PLUS; break;
1139
1140 case TokenKind::NEGLPAREN: type = GroupNode<charT>::GroupType::NEG; break;
1141
1142 case TokenKind::ATLPAREN: type = GroupNode<charT>::GroupType::AT; break;
1143
1144 default: throw Error("Not valid group"); break;
1145 }
1146
1147 AstNodePtr<charT> group_glob = ParserUnion();
1148 tk = NextToken();
1149 if (tk != TokenKind::RPAREN) {
1150 throw Error("Expected ')' at and of group");
1151 }
1152
1153 return AstNodePtr<charT>(new GroupNode<charT>(type, std::move(group_glob)));
1154 }
1155
1156 AstNodePtr<charT> ParserConcat()
1157 {
1158 auto check_end = [&]() -> bool {
1159 Token<charT> &tk = GetToken();
1160
1161 switch (tk.Kind()) {
1162 case TokenKind::EOS:
1163 case TokenKind::RPAREN:
1164 case TokenKind::UNION: return true;
1165
1166 default: return false;
1167 }
1168 };
1169
1170 std::vector<AstNodePtr<charT>> parts;
1171
1172 while (!check_end()) {
1173 parts.push_back(ParserBasicGlob());
1174 }
1175
1176 return AstNodePtr<charT>(new ConcatNode<charT>(std::move(parts)));
1177 }
1178
1179 AstNodePtr<charT> ParserUnion()
1180 {
1181 std::vector<AstNodePtr<charT>> items;
1182 items.push_back(ParserConcat());
1183
1184 while (GetToken() == TokenKind::UNION) {
1185 Advance();
1186 items.push_back(ParserConcat());
1187 }
1188
1189 return AstNodePtr<charT>(new UnionNode<charT>(std::move(items)));
1190 }
1191
1192 AstNodePtr<charT> ParserGlob()
1193 {
1194 AstNodePtr<charT> glob = ParserConcat();
1195
1196 if (GetToken() != TokenKind::EOS) {
1197 throw Error("Expected the end of glob");
1198 }
1199
1200 return AstNodePtr<charT>(new GlobNode<charT>(std::move(glob)));
1201 }

1216 {

1221 pos_++;

1223 }

1226 {
1227 if (pos_ == tok_vec_.size() - 1)
1228 return false;

1230 ++pos_;
1231 return true;
1232 }

1392 for (auto &item : items) {
1393 std::unique_ptr<Automata<charT>> automata_ptr(new Automata<charT>);
1394 AstConsumer ast_consumer;

1396

1398 automata_ptr->GetState(ast_consumer.preview_state_).AddNextState(match_state);
1399 automata_ptr->SetMatchState(match_state);
1400
1401 size_t fail_state = automata_ptr->template NewState<StateFail<charT>>();
1402 automata_ptr->SetFailState(fail_state);

1404 vec_automatas.push_back(std::move(automata_ptr));
1405 }
