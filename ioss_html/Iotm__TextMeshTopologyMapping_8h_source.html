27 struct TopologyMapEntry
28 {
29 using DimensionArray = bool[4];
30
31 unsigned int id;
32 Ioss::ElementTopology *topology;
33
34 // Defines what spatial dimension the topology is valid on
35 DimensionArray validSpatialDimensions;
36
37 TopologyMapEntry()
38 : id(Ioss::ElementTopology::get_unique_id(Ioss::Unknown::name)),
39 topology(Ioss::ElementTopology::factory(Ioss::Unknown::name))
40 {
41 set_valid_spatial_dimensions({false, false, false, false});
42 }
43
44 TopologyMapEntry(const std::string &name, const DimensionArray &validSpatialDimensions_)
45 : id(Ioss::ElementTopology::get_unique_id(name)),
46 topology(Ioss::ElementTopology::factory(name))
47 {
48 set_valid_spatial_dimensions(validSpatialDimensions_);
49 }
50
51 TopologyMapEntry(const TopologyMapEntry &topo) : id(topo.id), topology(topo.topology)
52 {
53 set_valid_spatial_dimensions(topo.validSpatialDimensions);
54 }
55
56 void set_valid_spatial_dimensions(const DimensionArray &validSpatialDimensions_)
57 {
58 validSpatialDimensions[0] = validSpatialDimensions_[0];
59 validSpatialDimensions[1] = validSpatialDimensions_[1];
60 validSpatialDimensions[2] = validSpatialDimensions_[2];
61 validSpatialDimensions[3] = validSpatialDimensions_[3];
62 }
63
64 bool defined_on_spatial_dimension(const unsigned spatialDim) const
65 {
66 if (spatialDim > 3) {
67 return false;
68 }
69 return validSpatialDimensions[spatialDim];
70 }
71
72 const std::string &name() const { return topology->name(); }
73
74 int num_nodes() const { return topology->number_nodes(); }
75
76 bool equivalent_valid_spatial_dimensions(const DimensionArray &validSpatialDimensions_) const
77 {
78 return validSpatialDimensions[0] == validSpatialDimensions_[0] &&
79 validSpatialDimensions[1] == validSpatialDimensions_[1] &&
80 validSpatialDimensions[2] == validSpatialDimensions_[2] &&
81 validSpatialDimensions[3] == validSpatialDimensions_[3];
82 }
83
84 bool operator==(const TopologyMapEntry &rhs) const
85 {
86 return id == rhs.id && topology == rhs.topology &&
87 equivalent_valid_spatial_dimensions(rhs.validSpatialDimensions);
88 }
89
90 bool operator!=(const TopologyMapEntry &rhs) const { return !(*this == rhs); }
91 };
