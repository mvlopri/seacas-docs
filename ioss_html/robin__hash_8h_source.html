151  /**
152  * Each bucket entry has:
153  * - A value of type `ValueType`.
154  * - An integer to store how far the value of the bucket, if any, is from its
155  * ideal bucket (ex: if the current bucket 5 has the value 'foo' and
156  * `hash('foo') % nb_buckets` == 3, `dist_from_ideal_bucket()` will return 2 as
157  * the current value of the bucket is two buckets away from its ideal bucket) If
158  * there is no value in the bucket (i.e. `empty()` is true)
159  * `dist_from_ideal_bucket()` will be < 0.
160  * - A marker which tells us if the bucket is the last bucket of the bucket
161  * array (useful for the iterator of the hash table).
162  * - If `StoreHash` is true, 32 bits of the hash of the value, if any, are also
163  * stored in the bucket. If the size of the hash is more than 32 bits, it is
164  * truncated. We don't store the full hash as storing the hash is a potential
165  * opportunity to use the unused space due to the alignment of the bucket_entry
166  * structure. We can thus potentially store the hash without any extra space
167  * (which would not be possible with 64 bits of the hash).
168  */

334  /**
335  * Internal common class used by `robin_map` and `robin_set`.
336  *
337  * ValueType is what will be stored by `robin_hash` (usually `std::pair<Key, T>`
338  * for map and `Key` for set).
339  *
340  * `KeySelect` should be a `FunctionObject` which takes a `ValueType` in
341  * parameter and returns a reference to the key.
342  *
343  * `ValueSelect` should be a `FunctionObject` which takes a `ValueType` in
344  * parameter and returns a reference to the value. `ValueSelect` should be void
345  * if there is no value (in a set for example).
346  *
347  * The strong exception guarantee only holds if the expression
348  * `std::is_nothrow_swappable<ValueType>\:\:value &&
349  * std::is_nothrow_move_constructible<ValueType>\:\:value` is true.
350  *
351  * Behaviour is undefined if the destructor of `ValueType` throws.
352  */

354  class Allocator, bool StoreHash, class GrowthPolicy>
355  class robin_hash : private Hash, private KeyEqual, private GrowthPolicy

357  private:
358  template <typename U>
359  using has_mapped_type = typename std::integral_constant<bool, !std::is_same<U, void>::value>;

390  StoreHash || ((sizeof(tsl::detail_robin_hash::bucket_entry<value_type, true>) ==
391  sizeof(tsl::detail_robin_hash::bucket_entry<value_type, false>)) &&
392  (sizeof(std::size_t) == sizeof(truncated_hash_type) ||

396  !std::is_same<Hash, std::hash<key_type>>::value));

405  /**
406  * We can only use the hash on rehash if the size of the hash type is the same
407  * as the stored one or if we use a power of two modulo. In the case of the
408  * power of two modulo, we just mask the least significant bytes, we just have
409  * to check that the truncated_hash_type didn't truncated more bytes.
410  */
411  static bool USE_STORED_HASH_ON_REHASH(size_type bucket_count)
412  {
413  if (STORE_HASH && sizeof(std::size_t) == sizeof(truncated_hash_type)) {
414  TSL_RH_UNUSED(bucket_count);
415  return true;
416  }
417  else if (STORE_HASH && is_power_of_two_policy<GrowthPolicy>::value) {
418  tsl_rh_assert(bucket_count > 0);
419  return (bucket_count - 1) <= std::numeric_limits<truncated_hash_type>::max();
420  }
421  else {
422  TSL_RH_UNUSED(bucket_count);
423  return false;
424  }

822  iterator erase(const_iterator first, const_iterator last)
823  {
824  if (first == last) {
825  return mutable_iterator(first);
826  }
827 
828  auto first_mutable = mutable_iterator(first);
829  auto last_mutable = mutable_iterator(last);
830  for (auto it = first_mutable.m_bucket; it != last_mutable.m_bucket; ++it) {
831  if (!it->empty()) {
832  it->clear();
833  m_nb_elements--;
834  }
835  }
836 
837  if (last_mutable == end()) {
838  m_try_shrink_on_next_insert = true;
839  return end();
840  }
841 
842  /*
843  * Backward shift on the values which come after the deleted values.
844  * We try to move the values closer to their ideal bucket.
845  */
846  std::size_t icloser_bucket = static_cast<std::size_t>(first_mutable.m_bucket - m_buckets);
847  std::size_t ito_move_closer_value =
848  static_cast<std::size_t>(last_mutable.m_bucket - m_buckets);
849  tsl_rh_assert(ito_move_closer_value > icloser_bucket);
850 
851  const std::size_t ireturn_bucket =
852  ito_move_closer_value -
853  std::min(ito_move_closer_value - icloser_bucket,
854  std::size_t(m_buckets[ito_move_closer_value].dist_from_ideal_bucket()));
855 
856  while (ito_move_closer_value < m_bucket_count &&
857  m_buckets[ito_move_closer_value].dist_from_ideal_bucket() > 0) {
858  icloser_bucket =
859  ito_move_closer_value -
860  std::min(ito_move_closer_value - icloser_bucket,
861  std::size_t(m_buckets[ito_move_closer_value].dist_from_ideal_bucket()));
862 
863  tsl_rh_assert(m_buckets[icloser_bucket].empty());
864  const distance_type new_distance =
865  distance_type(m_buckets[ito_move_closer_value].dist_from_ideal_bucket() -
866  (ito_move_closer_value - icloser_bucket));
867  m_buckets[icloser_bucket].set_value_of_empty_bucket(
868  new_distance, m_buckets[ito_move_closer_value].truncated_hash(),
869  std::move(m_buckets[ito_move_closer_value].value()));
870  m_buckets[ito_move_closer_value].clear();
871 
872  ++icloser_bucket;
873  ++ito_move_closer_value;
874  }
875 
876  m_try_shrink_on_next_insert = true;
877 
878  return iterator(m_buckets + ireturn_bucket);
879  }

1176  new_distance, m_buckets[ibucket].truncated_hash(),

1185  template <class K, class... Args>
1186  std::pair<iterator, bool> insert_impl(const K &key, Args &&...value_type_args)
1187  {
1188  const std::size_t my_hash = hash_key(key);
1189 
1190  std::size_t ibucket = bucket_for_hash(my_hash);
1191  distance_type dist_from_ideal_bucket = 0;
1192 
1193  while (dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {
1194  if ((!USE_STORED_HASH_ON_LOOKUP || m_buckets[ibucket].bucket_hash_equal(my_hash)) &&
1195  compare_keys(KeySelect()(m_buckets[ibucket].value()), key)) {
1196  return std::make_pair(iterator(m_buckets + ibucket), false);
1197  }
1198 
1199  ibucket = next_bucket(ibucket);
1200  dist_from_ideal_bucket++;
1201  }
1202 
1203  if (rehash_on_extreme_load()) {
1204  ibucket = bucket_for_hash(my_hash);
1205  dist_from_ideal_bucket = 0;
1206 
1207  while (dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {
1208  ibucket = next_bucket(ibucket);
1209  dist_from_ideal_bucket++;
1210  }
1211  }
1212 
1213  if (m_buckets[ibucket].empty()) {
1214  m_buckets[ibucket].set_value_of_empty_bucket(dist_from_ideal_bucket,
1215  bucket_entry::truncate_hash(my_hash),
1216  std::forward<Args>(value_type_args)...);
1217  }
1218  else {
1219  insert_value(ibucket, dist_from_ideal_bucket, bucket_entry::truncate_hash(my_hash),
1220  std::forward<Args>(value_type_args)...);
1221  }
1222 
1223  m_nb_elements++;
1224  /*
1225  * The value will be inserted in ibucket in any case, either because it was
