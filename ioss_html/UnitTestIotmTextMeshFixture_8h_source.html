60struct SideEntryLess
61{
62 inline bool operator()(const SideEntry &lhs, const SideEntry &rhs) const
63 {
64 if (lhs.first < rhs.first)
65 return true;
66 else if (lhs.first == rhs.first && lhs.second < rhs.second)
67 return true;
68
69 return false;
70 }
71};

413 template <typename INT>
414 ElementInfo get_element_info_from_block_impl(EntityId elemId,
415 const Ioss::ElementBlock *block) const
416 {
417 const Ioss::ElementTopology *topo = nullptr;
418 EntityIdVector elemConn;
419
420 std::vector<INT> connectivity;
421 std::vector<INT> elemIds;
422
423 block->get_field_data("ids", elemIds);
424 block->get_field_data("connectivity", connectivity);
425
426 topo = block->topology();
427
428 size_t elementCount = elemIds.size();
429 int nodesPerElem = topo->number_nodes();
430
431 for (size_t i = 0; i < elementCount; ++i) {
432 INT *conn = &connectivity[i * nodesPerElem];
433 EntityId id = static_cast<EntityId>(elemIds[i]);
434
435 if (id == elemId) {
436 for (int j = 0; j < nodesPerElem; j++) {
437 elemConn.push_back(conn[j]);
438 }
439
440 return ElementInfo(topo, elemConn);
441 }
442 }
443
444 return ElementInfo(nullptr, EntityIdVector());
445 }

637 void filename_substitution(std::string &filename)
638 {
639 // See if filename contains "%P" which is replaced by the number of processors...
640 // Assumes that %P only occurs once...
641 // filename is changed.
642 size_t pos = filename.find("%P");
643 if (pos != std::string::npos) {
644 // Found the characters... Replace with the processor count...
645 int num_proc = std::max(1, get_parallel_size());
646 std::string tmp(filename, 0, pos);
647 tmp += std::to_string(num_proc);
648 tmp += filename.substr(pos + 2);
649 filename = tmp;
650 }
651 }

653 std::pair<std::string, std::string>
654 get_database_type_and_filename(const std::string &meshDesc)
655 {
656 std::string type;
657 std::string filename;
658
659 size_t colon = meshDesc.find(':');
660 if (colon != std::string::npos && colon > 0) {
661 type = meshDesc.substr(0, colon);
662 filename = meshDesc.substr(colon + 1);
663 }
664 else {
665 type = "textmesh";
666 filename = meshDesc;
667 }
668
669 return std::make_pair(type, filename);
670 }
