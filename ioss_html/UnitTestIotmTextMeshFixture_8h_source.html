263 template <typename INT>
264 ElementInfo get_element_info_from_block_impl(EntityId elemId,
265 const Ioss::ElementBlock *block) const
266 {
267 const Ioss::ElementTopology *topo = nullptr;
268 EntityIdVector elemConn;
269
270 std::vector<INT> connectivity;
271 std::vector<INT> elemIds;
272
273 block->get_field_data("ids", elemIds);
274 block->get_field_data("connectivity", connectivity);
275
276 topo = block->topology();
277
278 size_t elementCount = elemIds.size();
279 int nodesPerElem = topo->number_nodes();
280
281 for (size_t i = 0; i < elementCount; ++i) {
282 INT *conn = &connectivity[i * nodesPerElem];
283 EntityId id = static_cast<EntityId>(elemIds[i]);
284
285 if (id == elemId) {
286 for (int j = 0; j < nodesPerElem; j++) {
287 elemConn.push_back(conn[j]);
288 }
289
290 return ElementInfo(topo, elemConn);
291 }
292 }
293
294 return ElementInfo(nullptr, EntityIdVector());
295 }

357 template <typename INT>
358 bool node_is_shared_with_proc_impl(EntityId nodeId, int sharingProc) const
359 {
360 ThrowRequireWithMsg(m_region != nullptr, "Ioss region has not been created");
361
362 Ioss::CommSet *io_cs = m_region->get_commset("commset_node");
363 size_t numSharings = io_cs->get_field("entity_processor").raw_count();
364
365 std::vector<INT> entityProc;
366 io_cs->get_field_data("entity_processor", entityProc);
367
368 for (size_t i = 0; i < numSharings; ++i) {
369 EntityId iossNodeId = entityProc[i * 2];
370 int iossSharingProc = entityProc[i * 2 + 1];
371
372 if (iossNodeId == nodeId && iossSharingProc == sharingProc) {
373 return true;
374 }
375 }
376
377 return false;
378 }

455 void filename_substitution(std::string &filename)
456 {
457 // See if filename contains "%P" which is replaced by the number of processors...
458 // Assumes that %P only occurs once...
459 // filename is changed.
460 size_t pos = filename.find("%P");
461 if (pos != std::string::npos) {
462 // Found the characters... Replace with the processor count...
463 int num_proc = std::max(1, get_parallel_size());
464 std::string tmp(filename, 0, pos);
465 tmp += std::to_string(num_proc);
466 tmp += filename.substr(pos + 2);
467 filename = tmp;
468 }
469 }

471 std::pair<std::string, std::string>
472 get_database_type_and_filename(const std::string &meshDesc)
473 {
474 std::string type;
475 std::string filename;
476
477 size_t colon = meshDesc.find(':');
478 if (colon != std::string::npos && colon > 0) {
479 type = meshDesc.substr(0, colon);
480 filename = meshDesc.substr(colon + 1);
481 }
482 else {
483 type = "textmesh";
484 filename = meshDesc;
485 }
486
487 return std::make_pair(type, filename);
488 }

502 {
503 verify_num_nodes();
504
505 for (size_t nodeIndex = 0; nodeIndex < goldNodeIds.size(); nodeIndex++) {

507 EXPECT_TRUE(is_valid_node(nodeId));
508
509 const double *nodalCoords = get_nodal_coordinates(nodeId);
510 const double *goldCoords = &goldCoordinates[nodeIndex * spatialDim];
511
512 verify_nodal_coordinates(nodeId, goldCoords, nodalCoords);

592 void fill_coordinates_from_ioss()
593 {
594 std::vector<double> iossCoordinates;
595
596 const Ioss::NodeBlockContainer &node_blocks = region.get_node_blocks();
597 assert(node_blocks.size() == 1);
598
599 Ioss::NodeBlock *nb = node_blocks[0];
600
601 size_t node_count = nb->get_property("entity_count").get_int();
602
603 EntityIdVector nodeIds = get_node_ids();
604
605 size_t numIossNodes =
606 field_data_from_ioss<double>(nb, "mesh_model_coordinates", iossCoordinates);
607 ThrowRequireWithMsg(node_count == numIossNodes, "Node count mismatch");
608 ThrowRequireWithMsg(iossCoordinates.size() == numIossNodes * spatialDim,
609 "Invalid coordinate data size");
610
611 fill_coordinate_map(nodeIds, iossCoordinates);
612 }
