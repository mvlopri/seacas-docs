71 inline std::vector<std::string> get_tokens(const std::string &str,
72 const std::string &separators)
73 {
74 std::vector<std::string> tokens;
75 auto first = std::begin(str);
76 while (first != std::end(str)) {
77 const auto second =
78 std::find_first_of(first, std::end(str), std::begin(separators), std::end(separators));
79 if (first != second) {
80 std::string token = strip(std::string(first, second));

125
126 if (name.length() < prefixLength + 1)
