117 const std::string blockPrefix = "BLOCK_";

303 bool has_token() const { return m_token != ""; }
304 bool has_newline() const { return m_token == "\n"; }
305 bool has_number() const { return has_token() && m_isNumber; }
306 bool has_string() const { return has_token() && !has_number() && !has_newline(); }
307
308 private:
309 void read_next_token()
310 {
311 m_oldToken = m_token;
312
313 if (char_is_newline()) {
314 m_isNumber = false;
315 m_token = "\n";
316 m_currentIndex++;
317 return;
318 }
319
320 m_token = "";
321 m_isNumber = true;
322
323 while (has_more_input()) {
324 if (char_is_whitespace()) {
325 m_currentIndex++;
326 continue;
327 }
328
329 if (char_is_comma()) {
330 m_currentIndex++;
331 break;
332 }
333
334 if (char_is_newline()) {
335 break;
336 }
337
338 m_isNumber &= char_is_digit();
339 m_token += current_char();
340 m_currentIndex++;
341 }
342 }
343
344 bool has_more_input() { return m_currentIndex < m_input.size(); }
345
346 bool char_is_whitespace() { return current_char() == ' '; }
347 bool char_is_comma() { return current_char() == ','; }
348 bool char_is_newline() { return current_char() == '\n'; }
349 bool char_is_digit() { return std::isdigit(static_cast<unsigned char>(current_char())); }
350
351 char current_char() { return m_input[m_currentIndex]; }
352
353 std::string make_upper_case(std::string str)
354 {
355 std::transform(str.begin(), str.end(), str.begin(), ::toupper);
356 return str;
357 }

360 unsigned m_currentIndex;
